<script>
function calculateSDLT(price, type) {
    let tax = 0;
    if (type === 'firstTime') {
        if (price <= 425000) tax = 0;
        else if (price <= 625000) tax = (price - 425000) * 0.05;
        else return calculateSDLT(price, 'standard'); // No relief over £625k
    } else {
        if (price > 1500000) { tax += (price - 1500000) * 0.12; price = 1500000; }
        if (price > 925000) { tax += (price - 925000) * 0.10; price = 925000; }
        if (price > 250000) { tax += (price - 250000) * 0.05; price = 250000; }
    }
    if (type === 'additional') tax += (parseFloat(document.getElementById('price').value) * 0.03);
    return tax;
}

function calculateAll() {
    const price = parseFloat(document.getElementById('price').value);
    const deposit = parseFloat(document.getElementById('deposit').value);
    const P = price - deposit; // Loan Amount
    const type = document.getElementById('buyerType').value;
    
    const totalYears = parseInt(document.getElementById('totalYears').value);
    const p1Years = parseInt(document.getElementById('p1Years').value);
    const r1 = parseFloat(document.getElementById('p1Rate').value) / 100 / 12;
    const r2 = parseFloat(document.getElementById('p2Rate').value) / 100 / 12;

    const totalMonths = totalYears * 12;
    const p1Months = p1Years * 12;

    // SDLT
    const sdlt = calculateSDLT(price, type);

    // Mortgage Logic
    const m1 = P * (r1 * Math.pow(1 + r1, totalMonths)) / (Math.pow(1 + r1, totalMonths) - 1);
    const balanceAfterP1 = (P * Math.pow(1 + r1, p1Months)) - (m1 * (Math.pow(1 + r1, p1Months) - 1) / r1);
    const m2 = balanceAfterP1 * (r2 * Math.pow(1 + r2, totalMonths - p1Months)) / (Math.pow(1 + r2, totalMonths - p1Months) - 1);

    const totalPaid = (m1 * p1Months) + (m2 * (totalMonths - p1Months));
    const totalInterest = totalPaid - P;

    // Format & Display
    const fmt = new Intl.NumberFormat('en-GB', { style: 'currency', currency: 'GBP' });
    document.getElementById('sdltResult').innerText = fmt.format(sdlt);
    document.getElementById('totalCash').innerText = fmt.format(sdlt + deposit);
    document.getElementById('m1').innerText = fmt.format(m1);
    document.getElementById('m2').innerText = fmt.format(m2);
    document.getElementById('totalInt').innerText = fmt.format(totalInterest);
    document.getElementById('grandTotal').innerText = fmt.format(totalPaid);
    
    document.getElementById('results').style.display = 'block';
}
</script>
